% print pages
\newcommand{\printListOfFigures}{
	\cleardoublepage
	\phantomsection
	\addcontentsline{toc}{chapter}{\listfigurename}
	\listoffigures
}

\newcommand{\printListOfTables}{
	\cleardoublepage
	\phantomsection
	\addcontentsline{toc}{chapter}{\listtablename}
	\listoftables
}

\newcommand{\printListOfAcronyms}{

	\phantomsection

\newcommand{\printListOfGlossaries}{
	\setlist[description]{style=nextline}
	\printglossary
	\setlist[description]{} % reset formatting of description lists
}

\newcommand{\enableHyperlinksForGlossariesAndAcronyms}{
	\acsetup{make-links} % enables hyperlinks for acronyms
	%glossary-hyperlinks are enabled by default
}

\newcommand{\disableHyperlinksForGlossariesAndAcronyms}{
	% acronym-hyperlinks are disabled by default
	\glsdisablehyper % disables hyperlinks for glossaries
}

\newcommand{\setPageNumberingToLowercaseRoman}{
	\pagenumbering{roman}
}

\newcommand{\setPageNumberingToArabicStartingFromOne}{
	\cleardoublepage{}
	\pagenumbering{arabic}
	\setcounter{page}{1}
}

\newcommand{\startAppendixIncludingRomanPageNumbering}{
	\cleardoublepage{}
	\appendix
	\pagenumbering{Roman}
}

%%% handle equation printing
\newcommand{\listequationsname}{Formelverzeichnis}
\newlistof{myequations}{equ}{\listequationsname}
\newcommand{\myequations}[1]{%
	\addcontentsline{equ}{myequations}{\protect\numberline{\theequation}#1}\par}
\setlength{\cftmyequationsnumwidth}{2.3em}
\setlength{\cftmyequationsindent}{1.5em}

\newcommand{\printListOfEquations}{
	\cleardoublepage
	\phantomsection
	\addcontentsline{toc}{chapter}{\listequationsname}
	\listofmyequations
}
